#include <BluetoothSerial.h>

BluetoothSerial SerialBT;

void setup() {
  pinMode(D1, OUTPUT);
  SerialBT.begin("DeneyapKart");
}

void loop() {
  if (SerialBT.available()) {
    char veri = SerialBT.read();
    if (veri == '1') {
      digitalWrite(D1, HIGH);
    }
    else if (veri == '0') {
      digitalWrite(D1, LOW);
    }
  }
}
