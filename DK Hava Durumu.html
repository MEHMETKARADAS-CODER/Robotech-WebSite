Arduino Kodu

#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <LiquidCrystal_I2C.h>

const char* ssid = "wifi";  // Wi-Fi ismi
const char* password = "sifre"; // Wi-Fi ≈üifresi

const char* server = "http://api.openweathermap.org/data/2.5/weather?q=Adana,tr&appid=7e46c1cd48cf48ecfb81239585f4ae07&units=metric&lang=tr";
// 'Adana' yerine kendi ≈üehriniz

LiquidCrystal_I2C lcd(0x27, 16, 2);

void setup() {
  Serial.begin(115200);
  lcd.begin();
  lcd.backlight();
  lcd.setCursor(0, 0);
  lcd.print("WiFi Baglaniyor");

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("WiFi Baglandi!");
  delay(1000);
  lcd.clear();
}

void loop() {
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(server);

    int httpCode = http.GET();
    if (httpCode > 0) {
      String payload = http.getString();
      DynamicJsonDocument doc(2048);
      DeserializationError error = deserializeJson(doc, payload);

      if (!error) {
        float sicaklik = doc["main"]["temp"];
        const char* durum = doc["weather"][0]["description"];

        lcd.clear();
        lcd.setCursor(0, 0);
        lcd.print("Sicaklik:");
        lcd.print(sicaklik, 1);
        lcd.print(" C");

        lcd.setCursor(0, 1);
        lcd.print(durum);
      } else {
        lcd.clear();
        lcd.setCursor(0, 0);
        lcd.print("JSON HATA");
      }
    } else {
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Baglanti Hatasi");
    }
    http.end();
  } else {
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("WiFi Yok");
  }

  delay(60000);
}
